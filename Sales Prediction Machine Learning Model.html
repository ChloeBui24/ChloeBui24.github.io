import numpy as np

# Function to check if a column contains percentage strings
def is_percent_column(series):
    # Check if the column is of object type and contains percentage strings
    return series.dtype == "object" and series.str.contains(r"^\d+\.?\d*%$", na=False).any()

# Identify percentage columns in the `fast` dataset
percent_cols = [col for col in fast.columns if is_percent_column(fast[col])]

# Convert percentage strings to numeric values (divide by 100)
for col in percent_cols:
    fast[col] = pd.to_numeric(fast[col].str.replace("%", "", regex=True), errors='coerce') / 100

# Display the modified dataset summary to confirm the changes
print(fast.info())

import pandas as pd
import numpy as np
from sklearn.feature_selection import VarianceThreshold
from sklearn.preprocessing import StandardScaler

# Step 1: Remove Columns with Zero Variance
selector = VarianceThreshold(threshold=0)
selector.fit(model_data)
non_constant_columns = model_data.columns[selector.get_support()]
model_data_filtered = model_data[non_constant_columns]

# Step 2: Remove Columns with NA Values
model_data_filtered = model_data_filtered.dropna(axis=1)

# Step 3: Preserve the Target Column
target = model_data_filtered["profit"]

# Step 4: Extract Predictors (excluding 'profit')
predictors = model_data_filtered.drop(columns=["profit"])

# Step 5: Compute Correlation Matrix
cor_matrix = predictors.corr()

# **Step 6: Identify Highly Correlated Features (Correlation > 0.9)**
threshold = 0.9
high_corr_pairs = []

for i in range(len(cor_matrix.columns)):
    for j in range(i):
        corr_value = cor_matrix.iloc[i, j]
        if abs(corr_value) > threshold:
            colname1 = cor_matrix.columns[i]
            colname2 = cor_matrix.columns[j]
            high_corr_pairs.append((colname1, colname2, corr_value))

# Display the highly correlated column pairs
print("\nHighly Correlated Pairs (|Correlation| > 0.9):")
for col1, col2, corr_value in high_corr_pairs:
    print(f"{col1} - {col2}: {corr_value:.2f}")

# Extract columns to drop
high_corr = list(set([col1 for col1, col2, _ in high_corr_pairs]))

# Step 7: Remove Highly Correlated Variables
model_data2 = predictors.drop(columns=high_corr)

# Step 8: Reattach the Target Column
model_data2["profit"] = target.reset_index(drop=True)

# Display the shape of the processed dataset
print("\nProcessed Dataset Shape:", model_data2.shape)

from sklearn.model_selection import train_test_split, cross_val_score
from sklearn.linear_model import LinearRegression
from sklearn.metrics import mean_squared_error, r2_score
from statsmodels.stats.outliers_influence import variance_inflation_factor
import statsmodels.api as sm

# Step 1: Data Preparation
X = model_data2.drop(columns=["profit"])
y = model_data2["profit"]

# Train-Test Split
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.2, random_state=505)

# Selected Features (from RFE step)
top_vars = selected_features  # Ensure this is defined in the previous step

# Subset the data with selected features
X_train = X_train[top_vars]
X_test = X_test[top_vars]

# Step 2: Check for Multicollinearity using VIF
print("Variance Inflation Factor (VIF) Analysis:")

# Add constant to the dataset
X_train_vif = sm.add_constant(X_train)

# Calculate VIF
vif_data = pd.DataFrame({
    "Feature": X_train_vif.columns,
    "VIF": [variance_inflation_factor(X_train_vif.values, i) for i in range(X_train_vif.shape[1])]
})

# Display VIF values
print(vif_data[vif_data["Feature"] != "const"])

# Step 3: Train Linear Regression Model using Cross-Validation
model = LinearRegression()

# Cross-validation
cv_scores = cross_val_score(model, X_train, y_train, cv=5, scoring='neg_root_mean_squared_error')
print(f"\nCross-Validation RMSE: {-np.mean(cv_scores):.4f}")

# Fit the final model on the entire training set
model.fit(X_train, y_train)

# Step 4: Predict on Test Set
y_pred = model.predict(X_test)

# Evaluation Metrics
rmse = np.sqrt(mean_squared_error(y_test, y_pred))
r2 = r2_score(y_test, y_pred)

print(f"\nTest RMSE: {rmse:.4f}")
print(f"Test R-squared: {r2:.4f}")
